# PhoneNumberInput

Control for country code selection and phone number input

## Usage

```typescript
import { PhoneNumberInput, CountryItem } from '@portkey/did-ui-react';
import { useState } from 'react';

const App: React.FC = () => {
  const [phoneNumber, setPhoneNumber] = useState<string>('');
  const [countryCode, setCountryCode] = useState<CountryItem | undefined>();

  return (
    <PhoneNumberInput
      iso={'United Kingdom'}
      countryList={[{ country: 'United Kingdom', code: '44', iso: 'GB' }]}
      phoneNumber={phoneNumber}
      onAreaChange={setCountryCode}
      onPhoneNumberChange={setPhoneNumber}
    />
  );
};

export default App;
```

## API

| Property            | Description                             | Type                      | Default  | Version |
| :------------------ | :-------------------------------------- | :------------------------ | :------- | :------ |
| iso                 | Currently selected country code address | string                    | -        |         |
| countryList         | List of country code                    | CountryItem[];            | -        |         |
| phoneNumber         | Phone number entered                    | string                    | -        |         |
| onAreaChange        | Callback of changing country code       | (v: CountryItem) => void; | -        |         |
| onPhoneNumberChange | Callback of entering phone number       | (v: string) => void;      | -        |         |
| onCancel            | Callback of cancelling                  | () => void                | -        |         |
